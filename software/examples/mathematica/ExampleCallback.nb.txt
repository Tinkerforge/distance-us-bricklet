Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="555"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
dus=NETNew["Tinkerforge.BrickletDistanceUS",uid,ipcon]
ipcon@Connect[host,port]

(* Set Period for distance callback to 1s (1000ms) *)
(* Note: The distance callback is only called every second if the *)
(*       distance has changed since the last call! *)
dus@SetDistanceCallbackPeriod[1000]

(* Callback function for distance value *)
DistanceCB[sender_,distance_]:=
 Print["Distance Value: "<>ToString[distance]]
AddEventHandler[dus@Distance,DistanceCB]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[dus]
ReleaseNETObject[ipcon]
